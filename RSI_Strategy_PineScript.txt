//@version=6
strategy(
     "RSI binaria long-only",
     overlay = true,
     pyramiding = 0,
     initial_capital = 100000,
     default_qty_type = strategy.percent_of_equity,
     default_qty_value = 100,
     process_orders_on_close = true
)

// ===== Inputs =====
rsiLength  = input.int(20, "RSI Length", minval = 1)
buyLevel   = input.float(30, "Nivel de compra", step = 0.1)
sellLevel  = input.float(70, "Nivel de venta", step = 0.1)
showBg     = input.bool(true, "Mostrar fondo")
showMarks  = input.bool(true, "Mostrar flechas")

// ===== RSI =====
rsiValue = ta.rsi(close, rsiLength)

// ===== Señales =====
// Compra cuando RSI cruza HACIA ABAJO el 30
buySignal  = ta.crossunder(rsiValue, buyLevel)

// Venta cuando RSI cruza HACIA ARRIBA el 70
sellSignal = ta.crossover(rsiValue, sellLevel)

// ===== Estado =====
inPosition = strategy.position_size > 0

// ===== Lógica binaria =====
if barstate.isconfirmed
    if buySignal and not inPosition
        strategy.entry(
             "Long",
             strategy.long,
             alert_message = "BUY | " + syminfo.ticker + " | RSI=" + str.tostring(rsiValue, "#.00")
        )

    if sellSignal and inPosition
        strategy.close(
             "Long",
             alert_message = "SELL | " + syminfo.ticker + " | RSI=" + str.tostring(rsiValue, "#.00"),
             immediately = true
        )

// ===== Visual =====
bgcolor(showBg ? (strategy.position_size > 0 ? color.new(color.green, 88) : color.new(color.red, 94)) : na)

plotshape(
     showMarks and buySignal and not inPosition,
     title = "BUY",
     style = shape.triangleup,
     location = location.belowbar,
     size = size.small,
     color = color.green,
     text = "BUY"
)

plotshape(
     showMarks and sellSignal and inPosition,
     title = "SELL",
     style = shape.triangledown,
     location = location.abovebar,
     size = size.small,
     color = color.red,
     text = "SELL"
)
